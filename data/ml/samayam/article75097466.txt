	function getmatchupdatejsondata() {
		//console.log("==getmatchupdatejsondata==");
	   	 
	   	 $.ajax({
             method: "GET",
             url: "https://s3-ap-southeast-1.amazonaws.com/navbharattimes/webappcricwidget.htm"
         })
         .done(function(json) {
             json = json.Calendar;
             var livecoverage = false;
             
             var matchExist = false;
             var matchIndex = 0;
             var matchCounter = 0;
             
             $('.cricketContainerData').html('');
             
             for (i = 0; i < json.length; i++) {
                 if ((json[i].live == 1|| json[i].live == 3 || json[i].live == 4)) {
                    
                    $(".miniscorecard").show();
                    livecoverage = true;
                     
                    //console.log("seriesname : "+ json[i].seriesname.toLowerCase()); 
                     
                    if((json[i].seriesname.toLowerCase() == 'indian premier league') || (json[i].seriesname.toLowerCase() == 'icc cricket world cup') || (json[i].teama_short == 'IND') || (json[i].teamb_short == 'IND')){
                       matchExist=true;
                       matchIndex=i;
                       //console.log("matchIndex : "+ matchIndex);
                       scorewidgetdata(json[matchIndex]);
                       matchCounter++;
                    }
                 
                 }
             
                if (livecoverage == false) {
                    $(".miniscorecard").hide();
                    clearInterval(notCntTimer);
                }
                
             }//end for loop
            
            //console.log("matchCounter :" + matchCounter);
            if(matchCounter>1){
              $('.cricketSliderArrow').show();
            }
           
        });
